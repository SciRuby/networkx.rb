def initialize(edges)
  @edges = edges
  @index = {}
  @n = edges.length
  @curr = 0
  edges.each_with_index { |(key, _value), idx| @index[idx] = key }
end

def move_to_next
  @temp = @curr
  @curr = (@curr + 1) % @n
  raise StopIteration if @temp == @n - 1
end
